COMEÇANDO

Primeiros Passos

Comece a usar o MQ Client em minutos com nossa documentação clara e exemplos práticos.

Instalação 

go get github.com/seu-repo/mq-client

Conexão Básica 

package main

import (
    "fmt"
    "log"
    "github.com/seu-repo/mq-client/client"
)

func main() {
    // Conectar ao servidor MQ
    mq, err := client.Dial("mq://user:password@localhost:8080")
    if err != nil {
        log.Fatal(err)
    }
    defer mq.Stop()

    fmt.Println("Conectado com ID:", mq.ID)
}

Instalação 

npm install mq-client
# ou
yarn add mq-client

Conexão Básica 

const { MQ } = require('mq-client');

async function main() {
    try {
        // Conectar ao servidor MQ
        const mq = await MQ.dial('mq://user:password@localhost:8080');
        console.log('Conectado com ID:', mq.ID);
        
        // Fechar conexão ao sair
        process.on('SIGINT', () => mq.stop());
    } catch (err) {
        console.error('Erro de conexão:', err);
    }
}

main();
EXEMPLOS

Padrões Comuns de Uso

Aprenda com exemplos práticos de implementações reais.

Pub/Sub

Padrão de publicação/assinatura para mensagens broadcast.

// Publisher
mq.Pub("notifications", "Nova mensagem recebida")

// Subscriber
mq.Sub("notifications", func(msg client.MQData) {
    fmt.Printf("Nova notificação: %s\n", msg.Payload)
})
// Publisher
mq.pub('notifications', 'Nova mensagem recebida');

// Subscriber
mq.sub('notifications', (msg) => {
    console.log('Nova notificação:', msg.payload);
});

Req/Res

Padrão requisição/resposta para comunicação síncrona.

// Servidor
mq.Service("math.square", func(msg client.MQData, reply func(err string, payload string)) {
    num, _ := strconv.Atoi(msg.Payload)
    reply("", strconv.Itoa(num*num))
})

// Cliente
res, err := mq.Req("math.square", "5", 2*time.Second)
if err != nil {
    log.Fatal(err)
}
fmt.Println("Resultado:", res) // 25
// Servidor
mq.service('math.square', (msg, reply) => {
    const num = parseInt(msg.payload);
    reply(null, (num * num).toString());
});

// Cliente
const result = await mq.req('math.square', '5', 2000);
console.log('Resultado:', result); // 25
